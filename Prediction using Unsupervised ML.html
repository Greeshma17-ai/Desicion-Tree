Implementing the Elbow Method¶

#Finding the number of clusters
from sklearn.cluster import KMeans

error= []
for i in range (1,11):
    kmeans = KMeans(n_clusters = i).fit(x)
    kmeans.fit(x)
    error.append(kmeans.inertia_)
